#Using different n_estimators for random forest regressor
score=[]
mxd=[]
for x in range(1,30):
    model = RandomForestRegressor(n_estimators=x)
    X_train, X_test, y_train, y_test = train_test_split(X, y, test_size= 0.2)
    model.fit(X_train,y_train)
    y_pred= model.predict(X_test)
    mae=mean_absolute_error(y_test,y_pred)
    mse=mean_squared_error(y_test,y_pred)
    score.append(mse)
    mxd.append(x)
scores=np.array(score)
ax = sns.lineplot(x = mxd, y = scores)

print(f'Best accuracy score |{min(scores)}| achieved at estimators: |{np.argmin(scores) + 1}|')

from sklearn.tree import DecisionTreeRegressor
accuracy_scores = []
max_depths = []
np.random.seed(99)
for max_depth in range(1, 20):
    model3 = DecisionTreeRegressor(max_depth = max_depth)
    model3.fit(X_train, y_train)

    test_prediction = model3.predict(X_test)
    test_accuracy = mean_squared_error(y_test, test_prediction)

    max_depths.append(max_depth)
    accuracy_scores.append(test_accuracy)

ax = sns.lineplot(x = max_depths, y = accuracy_scores)
ax.set(xlabel='Decision Tree Max Depth', ylabel='Accuracy Score')
print(f'Best accuracy score |{min(accuracy_scores)}| achieved at max depth |{np.argmin(accuracy_scores) + 1}|')

model3 = DecisionTreeRegressor(max_depth=max_depths[np.argmin(accuracy_scores)])
